001package comp3110.crossingDomain;
002
003/**
004 *
005 * Defines the interface that Crossing Hardware domains use to communicate with the Crossing
006 * domain. It represents assumptions that the Crossing domain makes about the events that are
007 * generated by a crossing hardware domain.
008 *
009 * This interface is part of the specification of an Explicit Bridge between the Crossing Domain and
010 * some other domain that provides access to the crossing hardware.
011 */
012
013 
014// This is the client side of the explicit bridge between the Crossing and
015// the Crossing hardware domains. This interface declares those methods by which the hardware
016// domain can communicate with the Crossing domain
017// 
018public interface CrossingToHardwareClientInterface {
019
020 /**
021  * Used by a crossing hardware domain to tell the Crossing domain that a crossingRequest
022  * event has occurred in the real world. 
023  *
024  */
025  public void processCrossingRequestEvent();
026  
027 /**
028  * Used by a crossing hardware domain to tell the Crossing domain that a setFailMode
029  * event has occurred in the real world. 
030  */
031  public void processSetFailModeEvent();
032  
033 /**
034  * Used by a crossing hardware domain to tell the Crossing domain that a setTrafficMode
035  * event has occurred in the real world. 
036  */
037  public void processSetTrafficModeEvent();
038  
039
040}
