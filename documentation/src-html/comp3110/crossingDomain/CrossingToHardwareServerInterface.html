001package comp3110.crossingDomain;
002
003/**
004 *
005 * Defines the interface that the Crossing domain uses to communicate with a Crossing
006 * Hardware domain. It represents a set of assumptions that the Crossing domain makes about some
007 * crossing hardware domain. 
008 *
009 * This interface is part of the specification of an Explicit Bridge between the Crossing Domain and
010 * some other domain that provides access to the crossing hardware.
011 */
012
013 
014// This is the server side of the explicit bridge between the crossing and
015// the Crossing hardware domains. This interface declares those methods by which the Crossing
016// domain can communicate with the hardware domain
017// 
018public interface CrossingToHardwareServerInterface {
019
020 /**
021  * Used by the Crossing domain to the real world that the walkOn event needs to be processed
022  *
023  */
024  public void processWalkOnEvent();
025
026 /**
027  * Used by the Crossing domain to the real world that the walkOff event needs to be processed
028  *
029  */
030  public void processWalkOffEvent();
031
032 /**
033  * Used by the Crossing domain to the real world that the dontWalkOn event needs to be processed
034  *
035  */
036  public void processDontWalkOnEvent();
037
038 /**
039  * Used by the Crossing domain to the real world that the dontWalkOff event needs to be processed
040  *
041  */
042  public void processDontWalkOffEvent();
043
044 /**
045  * Used by the Crossing domain to the real world that the fail event needs to be processed
046  *
047  */
048  public void processFailEvent();
049
050
051 /**
052  * Bridge operation used by the Crossing domain to control the real world traffic light
053  */
054  public void setGreen();
055
056 /**
057  * Bridge operation used by the Crossing domain to control the real world traffic light
058  */
059  public void setYellow();
060
061 /**
062  * Bridge operation used by the Crossing domain to control the real world traffic light
063  */
064  public void setRed();
065
066}
