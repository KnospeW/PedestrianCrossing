001  package comp3110;
002
003
004   import comp3110.crossingDomain.*;
005   import comp3110.simulatedCrossingHardwareDomain.*;
006   import comp3110.mechanismsDomain.*;
007
008/**
009 *
010 * This is the main program for the pedestrian crossing application. It creates each of the
011 * executable domains in the application (instances of {@link comp8120.crossingDomain.CrossingDomain
012 * CrossingDomain} and {@link comp8120.simulatedCrossingHardwareDomain.SimulatedCrossingHardwareDomain
013 * SimulatedCrossingHardwareDomain}) and connects them together with an explicit bridge. The domains
014 * are then initialised and the application event queue is started. The program will then start responding
015 * to exeternal events.
016 */
017
018
019   // Main programs are declared within a class that extends
020   // comp8120.mechanismsDomain.Application.
021   //
022   public class Crossing extends Application {
023   
024      /**
025       * The initialise() method creates and connects the executable domains
026       * of the application. It then starts the application event queue. 
027       *
028       * See the <A HREF="../src-html/comp8120/Crossing.html">source code</A> for commented implementation.
029       */
030      public void initialise() {
031         
032         // Create instances of the two executable domains involved in the
033         // microwave oven
034         //
035         CrossingDomain                  crossingDomain  = new CrossingDomain();
036         SimulatedCrossingHardwareDomain hardwareDomain  = new SimulatedCrossingHardwareDomain();
037         
038         // Establish the explicit bridge between the Microwave and Hardware domains
039         //
040         crossingDomain.setHardwareEE(hardwareDomain);
041         hardwareDomain.setCrossingEE(crossingDomain);
042         
043         // Initialise the two domains
044         //
045         hardwareDomain.initialise();
046         crossingDomain.initialise();
047      
048         // Start the event queue in order to start the processing
049         // of events
050         //
051         ApplicationEventQueue.start();
052      }
053   
054      /**
055       * The standard Java main method. It creates an instance of the enclosing class and
056       * calls it's initialise() method to setup and start the microwave application.
057       */
058      public static void main(String[] args) {
059         Crossing crossing = new Crossing();
060         crossing.initialise();
061      }
062   
063   }
