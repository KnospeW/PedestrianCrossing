001   package comp3110.mechanismsDomain;
002
003  /**
004   * All transitions between states in an xtUML state model are implemented as instances of the Transition
005   * class.
006   */
007   public class Transition {
008   
009      State fromState;
010      State toState;
011      int   event;
012   
013      /**
014      * Constructs a transition between two states that will occur in response to
015      * a given event.
016      *
017      * @param fromState the state from which the event will cause a transition
018      * @param toState the state to which the object is required to move
019      * @param event the ID of the event that will cause the transition
020      */
021      public Transition(State fromState, State toState, int event) {
022         this.fromState  = fromState;
023         this.toState    = toState;
024         this.event      = event;
025      }
026   
027      /**
028      * returns the from state of the transition
029      *
030      * @return the from state
031      */
032      public State getFromState() {
033         return fromState;
034      }
035   
036     /**
037      * returns the to state of the transition
038      *
039      * @return the to state
040      */
041      public State getToState() {
042         return toState;
043      }
044   
045     /**
046      * returns the ID of the event associated with the transition
047      *
048      * @return the event ID
049      */
050      public int getEvent() {
051         return event;
052      }
053   
054   }
