001   package comp3110.mechanismsDomain;
002
003   import java.util.Vector;
004   
005/**
006 *
007 * This class implements a simple queue used by 
008 * the {@link comp3110.mechanismsDomain.ApplicationEventQueue ApplicationEventQueue}
009 */
010   
011   class EventQueue {
012      Vector queue;
013   
014      public EventQueue(int size) {
015         queue = new Vector(size);
016      }
017   
018      public synchronized void queueEvent(EventInstance theEvent) {
019         queue.addElement((Object)theEvent);
020         notifyAll();
021      }
022   
023      public synchronized EventInstance getEvent(){
024         while (queue.isEmpty())
025            try {
026               wait();
027            }
028               catch(InterruptedException e) {
029               }
030         EventInstance theObject = (EventInstance)queue.firstElement();
031         queue.removeElementAt(0);
032         return theObject;
033      }
034   }
