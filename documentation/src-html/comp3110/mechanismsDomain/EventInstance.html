001   package comp3110.mechanismsDomain;
002
003/**
004* All events generated during the execution of an application are implemented
005* as instances of the EventInstance Java class. This class is not often used directly by the programmer.
006*/
007
008   public class EventInstance {
009   
010      int           event;
011      Object        parameters;
012      ActiveClass  target;
013   
014      /**
015      * Creates a new EventInstance
016      *
017      * @param theEvent the event to be transmitted
018      * @param theParameters an object containing any event data
019      * @param theTarget the class that is to receive the event
020      */
021      public EventInstance(int theEvent, Object theParameters, ActiveClass theTarget) {
022         event      = theEvent;
023         parameters = theParameters;
024         target     = theTarget;
025      }
026   
027   /**
028   * returns the event associated with this EventInstance
029   *
030   * @return the event associated with this instance
031   */
032      public int getEvent() {
033         return event;
034      }
035   
036   /**
037   * returns the parameters associated with this EventInstance
038   *
039   * @return the parameters associated with this instance
040   */
041      public Object getParameters() {
042         return parameters;
043      }
044   
045   /**
046   * returns the target class associated with this EventInstance
047   *
048   * @return the target class associated with this instance
049   */
050   
051      public ActiveClass getTarget() {
052         return target;
053      }
054   
055   }
