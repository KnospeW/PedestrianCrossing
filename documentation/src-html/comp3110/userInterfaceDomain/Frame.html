001   package comp3110.userInterfaceDomain;
002
003   import java.awt.Component;
004   import java.awt.FlowLayout;
005
006/**
007* All User Interface domain application windows are implemented as classes which extend the Frame class.
008* The Frame class is implemented as a {@link javax.swing.JFrame javax.swing.JFrame} with an associated window event listener which closes
009* the entire application when the window is closed. The layout of the frame is based on
010* {@link java.awt.FlowLayout java.awt.FlowLayout}.
011*/
012
013   public class Frame extends javax.swing.JFrame {
014   
015       public Frame(String title, int width, int height) {
016         super(title);
017         setSize(width, height);
018         setVisible(true);
019         getContentPane().setLayout(new FlowLayout());
020      
021         addWindowListener(
022               new java.awt.event.WindowAdapter() {
023                  public void windowClosing(java.awt.event.WindowEvent e) {
024                     System.exit(0);
025                  }
026               });
027      }
028   
029      /**
030      * adds a specified component to the frame
031      *
032      * @param component the component (a PushButton or TextInputField) to be added to the frame
033      * @param position  the position of the component in the border layout (NORTH, SOUTH, EAST, WEST or CENTER)
034      */
035      public void addComponent(Component component) {
036         getContentPane().add(component);
037      }
038   
039      public void refresh() {
040         pack();
041      }
042   }
