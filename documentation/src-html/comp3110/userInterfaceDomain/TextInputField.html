001   package comp3110.userInterfaceDomain;
002
003   /**
004   * All GUI domain text fields are implemented as instances of the TextInputField class. Text Fields
005   * are extensions javax.swing.JTextField 
006   */
007
008   public class TextInputField extends javax.swing.JTextField {
009      
010      /**
011      * Constructs a TextInputField
012      *
013      * @param defaultText text that will be displayed in the TextInputField upon creation.
014      */
015      public TextInputField(String defaultText) {
016         super();
017         setPreferredSize(new java.awt.Dimension(50, 25));
018         setText(defaultText);
019         setVisible(true);
020      }
021   
022   }
