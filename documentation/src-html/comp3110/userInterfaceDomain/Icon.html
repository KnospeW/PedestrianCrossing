001   package comp3110.userInterfaceDomain;
002
003   /**
004   * All GUI domain buttons are implemented as instances of the PushButton class. Push Buttons
005   * are implemented as javax.swing.JButtons and are created with an instance of a class
006   * that implements the {@link comp3110.userInterfaceDomain.PushButtonCallback PushButtonCallback}
007   * interface. When a Push Button is clicked, the 'buttonPressed()' method of this object is called. 
008   */
009
010   public class Icon extends javax.swing.JLabel {
011   
012      private PushButtonCallback callback;
013   
014      /**
015      * Constructs a PushButton
016      *
017      * @param label the text to be displayed on the button
018      * @param callback an object that implements the {@link comp3110.userInterfaceDomain.PushButtonCallback PushButtonCallback} interface
019      */
020      public Icon(String filename) {
021        super();
022        setIcon(new javax.swing.ImageIcon(filename));
023      }
024
025
026     public void showIcon() {
027        enable();
028        repaint();
029     }
030   
031     public void hideIcon() {
032        disable();
033        repaint();
034     }
035   
036 
037   
038   }
